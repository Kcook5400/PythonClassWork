from Exceptions import CustIDError


class Customer:

    def __init__(self, last_name, first_name, phone_number, customer_id):
        self.last_name = last_name
        self.first_name = first_name
        self.phone_number = phone_number
        self.customer_id = customer_id
        try:
            customer_id in range(1000, 9999)
        except:
            raise Exception(CustIDError)
        try:
            last_name.isalpha()
        except:
            raise LastNameError

    def display(self):
        return str(self)

    def __str__(self):
        return 'Customer Id: ' + str(
            self.customer_id) + ', first name ' + self.first_name + ', last name ' + self.last_name + ', Phone:' + str(
            self.phone_number)
